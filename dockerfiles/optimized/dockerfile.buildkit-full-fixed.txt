        -DCMAKE_CXX_COMPILER_LAUNCHER=ccache \

# Build with all available cores using cache mount for ccache
WORKDIR /root/slicer/Slicer-SuperBuild-Release
RUN --mount=type=cache,target=/cache/ccache \
    make -j$(nproc)

# Package Slicer
RUN cd Slicer-build && \
    make package && \
    mkdir -p /root/slicer/packages && \
    mv *.tar.gz /root/slicer/packages/

RUN cd /root/slicer/modules && \

# Patch VTK_to_DICOMTract.cxx to replace CODE_DCM_* macros that were removed
# in newer versions of DCMTK with their explicit DSRBasicCodedEntry equivalents.
# DCMTK CID 7261 codes: 113211=Deterministic, 113212=Probabilistic,
#                        113213=Global, 113217=Bootstrap
RUN sed -i \
    -e 's/CODE_DCM_Deterministic/DSRBasicCodedEntry("113211", "DCM", "Deterministic")/g' \
    -e 's/CODE_DCM_Probabilistic/DSRBasicCodedEntry("113212", "DCM", "Probabilistic")/g' \
    -e 's/CODE_DCM_Global/DSRBasicCodedEntry("113213", "DCM", "Global")/g' \
    -e 's/CODE_DCM_Bootstrap/DSRBasicCodedEntry("113217", "DCM", "Bootstrap")/g' \
    /root/slicer/modules/SlicerDMRI/Modules/Loadable/TractIO/TractIOCLI/VTK_to_DICOMTract.cxx

RUN cd /root/slicer/modules && \
    mkdir SlicerDMRI-build && \
    cd SlicerDMRI-build && \
    cmake \
        -DCMAKE_BUILD_TYPE:STRING=Release \
        -DSlicer_DIR:PATH=/root/slicer/Slicer-SuperBuild-Release/Slicer-build \
        -DCMAKE_C_COMPILER_LAUNCHER=ccache \
        -DCMAKE_CXX_COMPILER_LAUNCHER=ccache \
        ../SlicerDMRI

RUN --mount=type=cache,target=/cache/ccache \
    cd /root/slicer/modules/SlicerDMRI-build && \
    make -j$(nproc) && \
    cd inner-build && \
    make package && \
    mv *.tar.gz /root/slicer/packages/

# Clone slicer_ros2_module plus source-only deps that have no ros-humble-* apt binary:
#   - object_recognition_msgs: wg-perception, ros2 branch  
#   - ros2_cisst_msgs: jhu-cisst, needed by slicer_ros2_module on Humble
RUN mkdir -p /root/ros2_ws/src \
    && cd /root/ros2_ws/src \
    && git clone -b devel https://github.com/rosmed/slicer_ros2_module \
    && git clone -b ros2 https://github.com/wg-perception/object_recognition_msgs \
    && git clone https://github.com/jhu-cisst/ros2_cisst_msgs

# Fix for ROS 2 Humble: Remove rosbag2 service clients that don't exist in Humble
# (Play, Stop, SplitBagFile are not available in ros-humble-rosbag2-interfaces)
RUN cd /root/ros2_ws/src/slicer_ros2_module/MRML \
    && sed -i '/rosbag2_interfaces\/srv\/Play/d' CMakeLists.txt \
    && sed -i '/rosbag2_interfaces\/srv\/SplitBagfile/d' CMakeLists.txt \
    && sed -i '/rosbag2_interfaces\/srv\/Stop/d' CMakeLists.txt

# Install additional ROS dependencies required by SlicerROS2

    ros-humble-moveit-msgs \
    ros-humble-object-recognition-msgs \
    && rm -rf /var/lib/apt/lists/*

# Run rosdep to install any remaining dependencies
RUN /bin/bash -c 'source /opt/ros/humble/setup.sh \
    && rosdep init 2>/dev/null || true \
    && rosdep update \
    && cd /root/ros2_ws \
    && rosdep install --from-paths src --ignore-src -r -y --rosdistro humble'

# Build the full workspace with colcon
# Note: Slicer_DIR points to the Slicer-build directory inside SuperBuild
RUN --mount=type=cache,target=/cache/ccache \
    /bin/bash -c 'source /opt/ros/humble/setup.sh \
    && cd /root/ros2_ws \
    && colcon build \
        --cmake-args \
            -DSlicer_DIR:PATH=/root/slicer/Slicer-SuperBuild-Release/Slicer-build \
            -DCMAKE_BUILD_TYPE=Release \
            -DCMAKE_C_COMPILER_LAUNCHER=ccache \
            -DCMAKE_CXX_COMPILER_LAUNCHER=ccache'

# The colcon build automatically installs the module files into the Slicer build directory

# Install Slicer modules (excluding SlicerTMS which is installed separately)
RUN cd /root/slicer/packages && \
    SLICER_DIR=$(find /root/slicer/packages -maxdepth 1 -type d -name "Slicer-*" | head -1) && \
    echo "=== Installing extensions to: ${SLICER_DIR} ===" && \
    for module in *SlicerOpenIGTLink*.tar.gz *SlicerIGSIO*.tar.gz *SlicerIGT*.tar.gz *ParallelProcessing*.tar.gz *SlicerDMRI*.tar.gz; do \
        if [ -f "$module" ]; then \
            echo "=== Installing $module ===" && \
            tar -xzf "$module" -C "${SLICER_DIR}" --strip-components=1 && \
            echo "=== Successfully installed $module ===" ; \
        fi \
    done

# Install SlicerROS2 loadable module into Slicer
RUN SLICER_DIR=$(find /root/slicer/packages -maxdepth 1 -type d -name "Slicer-*" | head -1) && \
    echo "=== Installing SlicerROS2 module ===" && \
    mkdir -p "${SLICER_DIR}/lib/Slicer-5.8/qt-loadable-modules" && \
    /bin/bash -c 'source /opt/ros/humble/setup.bash && \
        source /root/ros2_ws/install/setup.bash && \
        ROS2_MODULE_DIR=$(find /root/ros2_ws/install -path "*/lib/Slicer-5.8/qt-loadable-modules" -type d | head -1) && \
        if [ -n "$ROS2_MODULE_DIR" ]; then \
            cp -r "$ROS2_MODULE_DIR"/. '"${SLICER_DIR}"'/lib/Slicer-5.8/qt-loadable-modules/ && \
            echo "=== SlicerROS2 installed from $ROS2_MODULE_DIR ==="; \
        else \
            echo "WARNING: SlicerROS2 module dir not found — check colcon build output"; \
        fi'

# SOLUTION: Multiple approaches to ensure environment variables are available
# ==============================================================================

# Set environment variables in Dockerfile (base level)
ENV TMS_SERVER_HOST=tmsserver
ENV TMS_SERVER_PORT_1=18944
ENV TMS_SERVER_PORT_2=18945

# Approach 1: Add to system-wide profile (loaded by login shells)
RUN echo 'export TMS_SERVER_HOST="${TMS_SERVER_HOST:-tmsserver}"' >> /etc/profile && \
    echo 'export TMS_SERVER_PORT_1="${TMS_SERVER_PORT_1:-18944}"' >> /etc/profile && \
    echo 'export TMS_SERVER_PORT_2="${TMS_SERVER_PORT_2:-18945}"' >> /etc/profile

# Approach 2: Add to bash.bashrc (loaded by non-login interactive shells)
RUN echo 'export TMS_SERVER_HOST="${TMS_SERVER_HOST:-tmsserver}"' >> /etc/bash.bashrc && \
    echo 'export TMS_SERVER_PORT_1="${TMS_SERVER_PORT_1:-18944}"' >> /etc/bash.bashrc && \
    echo 'export TMS_SERVER_PORT_2="${TMS_SERVER_PORT_2:-18945}"' >> /etc/bash.bashrc

# Approach 3: Add to environment file (loaded by pam_env)
RUN echo "TMS_SERVER_HOST=tmsserver" >> /etc/environment && \
    echo "TMS_SERVER_PORT_1=18944" >> /etc/environment && \
    echo "TMS_SERVER_PORT_2=18945" >> /etc/environment

# Approach 4: Add to Kasm startup script so GUI apps inherit env variables
RUN echo 'export TMS_SERVER_HOST="${TMS_SERVER_HOST:-tmsserver}"' >> /dockerstartup/kasm_default_profile.sh && \
    echo 'export TMS_SERVER_PORT_1="${TMS_SERVER_PORT_1:-18944}"' >> /dockerstartup/kasm_default_profile.sh && \
    echo 'export TMS_SERVER_PORT_2="${TMS_SERVER_PORT_2:-18945}"' >> /dockerstartup/kasm_default_profile.sh
